let screen = document.getElementById('screen') let buttons = document.querySelectorAll('button'); let screenvalue = ""; for (items of buttons) { items.addEventListener('click', (e) => { buttonText = e.target.innerText; if (buttonText == '=') { screen.value = eval(screenvalue) } else if (buttonText == 'C') { screenvalue = ""; screen.value = ""; } else { screenvalue += buttonText screen.value = screenvalue } }) }
